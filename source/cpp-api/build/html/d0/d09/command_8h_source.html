46 //

47 // Binary function objects to be used by the find_if algorithm.

48 // The find_if algo. want to have a predicate, this means that the return value

49 // must be a boolean (R is its name).

50 // The find_if algo. needs a unary predicate. This function object is a binary

51 // function object. It must be used with the bind2nd function adapter

52 //

53

54 template <typename A1,typename A2,typename R>

55 struct WantedCmd : public binary_function<A1,A2,R>

56 {

57 R operator()(A1 cmd_ptr, A2 name) const

58  {

59  if (::strlen(name) != cmd_ptr->get_lower_name().size())

60  return false;

61  string tmp_name(name);

62 transform(tmp_name.begin(),tmp_name.end(),tmp_name.begin(),::tolower);

63  return cmd_ptr->get_lower_name() == tmp_name;

64  }

65 };

66 #endif
