1102 {

1110  string::size_type pos;

1114  print_usage(argv[0]);

1119  }

1120  if (arg_nb == argc)

1121  {

1122  cerr << “Missing ORB endPoint specification” << endl;

1123  print_usage(argv[0]);

1124  }
